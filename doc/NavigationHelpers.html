  def path_to(page_name)
    case page_name

    when %r^the (RottenPotatoes )?home\s?page$/ then '/movies'
    when %r^the Create New Movie page$/ then '/movies/new'
    when %rthe details page for "(.+)"/ then current_path
# the details page for "Star Wars"  
    when %rthe Similar Movies page for "(.+)"/ then current_path
#the Similar Movies page for "Star Wars"
    when %r^the Edit Existing Movie$/ then '/movie/edit'

    when %r^the home\s?page$/
      '/'

    # Add more mappings here.
    # Here is an example that pulls values out of the Regexp:
    #
   when %r^(.*)'s profile page$/
     user_profile_path(User.find_by_login($1))

    else
      begin
        page_name =~ %r^the (.*) page$/
        path_components = $1.split(%r\s+/)
        self.send(path_components.push('path').join('_').to_sym)
      rescue NoMethodError, ArgumentError
        raise "Can't find mapping from \"#{page_name}\" to a path.\n" +
          "Now, go and add a mapping in #{__FILE__}"
      end
    end
  end
